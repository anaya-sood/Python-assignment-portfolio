import math
import numpy as np


def roots(f, a, b, tolerance=1e-10, max_iterations=1000):
    """ 
    Find a root of function f on interval [a,b] using the bisection method. 
 
    Args: 
        f: The function to find roots for 
        a: Left endpoint of interval 
        b: Right endpoint of interval 
        tolerance: Desired accuracy (default 1e-10 for 10 decimal places) 
        max_iterations: Maximum number of iterations 
 
    Returns: 
        float: Approximate root of f(x) = 0 
    """
    # Check if endpoints have different signs
    if f(a) * f(b) > 0:
        return None  # No guaranteed root in interval

    for i in range(max_iterations):
        c = (a + b) / 2  # Midpoint

        if abs(f(c)) < tolerance:  # Found root within tolerance
            return c

        if f(c) * f(a) < 0:  # Root is in left half
            b = c
        else:  # Root is in right half
            a = c

    return (a + b) / 2  # Return best approximation if max iterations reached


# Test functions
def f1(x): return math.exp(x) + math.log(x)


def f2(x): return math.atan(x) - x ** 2


def f3(x): return math.sin(x) * math.log(x)


def f4(x): return math.log(math.cos(x))


# Run test cases
test_cases = [
    (f1, 0.1, 1),  # Modified slightly since ln(0) is undefined
    (f2, 0, 2),
    (f3, 3, 4),
    (f4, 5, 7)
]

print("Testing root finding for various functions:")
for f, a, b in test_cases:
    root = roots(f, a, b)
    if root is not None:
        print(f"\nRoot found at x ≈ {root:.10f}")
        print(f"f(x) ≈ {f(root):.10e}")
    else:
        print("\nNo root found in the given interval")
