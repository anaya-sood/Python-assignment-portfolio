def taylor_arctan(x):
    if x < 0 or x > 1:
        return "Error!"

    approximation = 0
    n = 0
    term = x  # Initial term for the series
    error_bound = 0.0001  # Desired error bound
    while abs(term) > error_bound:
        term = ((-1) ** n * x ** (2 * n + 1)) / (2 * n + 1)
        approximation += term
        n += 1

    return approximation, n
