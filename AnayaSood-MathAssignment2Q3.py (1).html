import numpy as np
import matplotlib.pyplot as plt


def funny(x):
    """ 
    Define the function: 
    f(x) = 0 for x <= -1 or x >= 1 
    f(x) = exp(1-1/(1-x^(10^250))) for -1 < x < 1 
    """
    if x <= -1 or x >= 1:
        return 0
    else:
        return np.exp(1 - (1 / (1 - x ** (10 ** 250))))


# Create arrays of x values for plotting
# We'll need to be careful near the discontinuities
x1 = np.linspace(-2, -1,1000)  # Values before -1

x4 = np.linspace(0.9, 0.999, 1000)  # Values just before 1
x5 = np.linspace(1, 2, 1000)  # Values after 1

# Calculate corresponding y values
y1 = [funny(x) for x in x1]

y3 = [funny(x) for x in x3]
y4 = [funny(x) for x in x4]

# Numerical analysis of limits
def analyze_limit(direction, point):
    """ 
    Analyze the limit of the function as x approaches a point from given direction 
    direction: 'left' or 'right' 
    point: the x-value being approached 
    """
    if direction == 'left':
        x_values = [point - 10 ** (-n) for n in range(1, 5)]
    else:
        x_values = [point + 10 ** (-n) for n in range(1, 5)]

    print(f"\nAnalyzing limit as x approaches {point} from the {direction}:")
    print("x\t\tf(x)")
    print("-" * 30)
    for x in x_values:
        try:
            y = funny(x)
            print(f"{x:.6f}\t{y:.6f}")
        except OverflowError:
            print(f"{x:.6f}\tOverflow")


# Analyze limits near discontinuities
analyze_limit('left', 1)
analyze_limit('right', -1)
